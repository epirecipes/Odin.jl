# Stochastic Discrete-Time SIR


## Introduction

This vignette demonstrates a stochastic discrete-time SIR model using
the R odin2/dust2 packages.

## Model Definition

``` r
library(odin2)
library(dust2)

sir_stoch <- odin({
  update(S) <- S - n_SI
  update(I) <- I + n_SI - n_IR
  update(R) <- R + n_IR
  initial(S) <- N - I0
  initial(I) <- I0
  initial(R) <- 0

  p_SI <- 1 - exp(-beta * I / N * dt)
  p_IR <- 1 - exp(-gamma * dt)
  n_SI <- Binomial(S, p_SI)
  n_IR <- Binomial(I, p_IR)

  beta <- parameter(0.5)
  gamma <- parameter(0.1)
  I0 <- parameter(10)
  N <- parameter(1000)
})
```

## Multiple Realisations

``` r
pars <- list(beta = 0.5, gamma = 0.1, I0 = 10, N = 1000)
n_particles <- 20
sys <- System(sir_stoch, pars, n_particles = n_particles, dt = 1, seed = 42)
dust_system_set_state_initial(sys)
times <- seq(0, 200, by = 1)
result <- simulate(sys, times)
cat("Result dimensions:", dim(result), "\n")
```

    Result dimensions: 3 20 201 

``` r
cols <- adjustcolor("red", alpha.f = 0.3)
plot(NULL, xlim = range(times), ylim = c(0, max(result[2, , ])),
     xlab = "Time", ylab = "Infected",
     main = "Stochastic SIR: 20 Realisations")
for (i in seq_len(n_particles)) {
  lines(times, result[2, i, ], col = cols)
}
```

## Comparing to Deterministic

``` r
sir_det <- odin({
  deriv(S) <- -beta * S * I / N
  deriv(I) <- beta * S * I / N - gamma * I
  deriv(R) <- gamma * I
  initial(S) <- N - I0
  initial(I) <- I0
  initial(R) <- 0
  beta <- parameter(0.5)
  gamma <- parameter(0.1)
  I0 <- parameter(10)
  N <- parameter(1000)
})
```
